<script setup>
import { onMounted } from 'vue'

function buildResults() {
  var lines = [];
  lines.push('HANDLED.SH WEBSITE REVIEW');
  lines.push('Date: ' + new Date().toLocaleDateString());
  lines.push('========================\n');

  document.querySelectorAll('.rv-section').forEach(function (section) {
    var title = section.querySelector('h2').textContent;
    lines.push(title.toUpperCase());
    lines.push('-'.repeat(title.length));

    section.querySelectorAll('.rv-q').forEach(function (q) {
      var label = q.querySelector('label').textContent;
      var stars = q.querySelector('.rv-stars');
      var rating = stars ? stars.getAttribute('data-selected') : null;
      var textarea = q.querySelector('textarea');
      var input = q.querySelector('input');
      var comment = textarea ? textarea.value : (input ? input.value : '');

      lines.push('\n' + label);
      if (rating) lines.push('Rating: ' + rating + '/5');
      if (comment) lines.push('Comment: ' + comment);
    });
    lines.push('');
  });

  return lines.join('\n');
}

if (typeof window !== 'undefined') {
  window.copyResults = function () {
    var text = buildResults();
    navigator.clipboard.writeText(text).then(function () {
      var toast = document.getElementById('rv-toast');
      toast.classList.add('show');
      setTimeout(function () { toast.classList.remove('show'); }, 2000);
    });
  };
}

onMounted(() => {
  var sections = document.querySelectorAll('.rv-section');
  var total = sections.length;
  var current = 0;

  var progressBar = document.getElementById('rv-progress-bar');
  var progressText = document.getElementById('rv-progress-text');
  var prevBtn = document.getElementById('rv-prev');
  var nextBtn = document.getElementById('rv-next');
  var submitBtn = document.getElementById('rv-submit');
  var submitAlt = document.getElementById('rv-submit-alt');

  // Collapse optional comment textareas
  document.querySelectorAll('.rv-q textarea[placeholder*="اختياري"]').forEach(function (ta) {
    var details = document.createElement('details');
    details.className = 'rv-comment';
    var summary = document.createElement('summary');
    summary.textContent = '+ إضافة تعليق';
    details.appendChild(summary);
    ta.parentNode.insertBefore(details, ta);
    details.appendChild(ta);
    if (ta.value.trim()) details.open = true;
  });

  // Build star ratings
  document.querySelectorAll('.rv-stars').forEach(function (container) {
    container.innerHTML = '';
    for (var i = 1; i <= 5; i++) {
      var star = document.createElement('button');
      star.type = 'button';
      star.className = 'rv-star';
      star.setAttribute('data-value', i);
      star.textContent = i;
      star.addEventListener('click', function (e) {
        var val = parseInt(e.target.getAttribute('data-value'));
        container.setAttribute('data-selected', val);
        container.querySelectorAll('.rv-star').forEach(function (s) {
          var sv = parseInt(s.getAttribute('data-value'));
          s.classList.toggle('active', sv <= val);
        });
      });
      container.appendChild(star);
    }
  });

  function showPage(idx) {
    sections.forEach(function (s, i) {
      s.style.display = i === idx ? '' : 'none';
    });
    current = idx;

    // Update progress
    var pct = Math.round(((idx + 1) / total) * 100);
    progressBar.style.width = pct + '%';
    progressText.textContent = (idx + 1) + ' / ' + total;

    // Update buttons
    prevBtn.style.display = idx === 0 ? 'none' : '';
    if (idx === total - 1) {
      nextBtn.style.display = 'none';
      submitBtn.style.display = '';
      submitAlt.style.display = '';
    } else {
      nextBtn.style.display = '';
      submitBtn.style.display = 'none';
      submitAlt.style.display = 'none';
    }

    // Scroll to top of form
    var progress = document.getElementById('rv-progress');
    if (progress) progress.scrollIntoView({ behavior: 'smooth', block: 'start' });
  }

  prevBtn.addEventListener('click', function () {
    if (current > 0) showPage(current - 1);
  });

  nextBtn.addEventListener('click', function () {
    if (current < total - 1) showPage(current + 1);
  });

  // Show first page
  showPage(0);

  // Form submit
  document.getElementById('reviewForm').addEventListener('submit', async function (e) {
    e.preventDefault();
    submitBtn.disabled = true;
    submitBtn.textContent = 'جارٍ الإرسال...';
    var status = document.getElementById('rv-status');
    status.textContent = '';
    status.className = 'rv-status';

    var body = buildResults();
    try {
      var resp = await fetch('https://app.handled.sh/api/website-review', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ review: body }),
      });
      if (!resp.ok) throw new Error('Request failed');
      document.getElementById('reviewForm').innerHTML = '<div class="rv-confirmation"><div class="rv-confirmation-icon">&#10003;</div><h3>شكرًا لك!</h3><p>تم إرسال مراجعتك. نحن نقدر ملاحظاتك حقًا.</p></div>';
    } catch (err) {
      status.textContent = 'حدث خطأ ما. يرجى المحاولة مرة أخرى أو استخدام "نسخ إلى الحافظة".';
      status.classList.add('error');
      submitBtn.disabled = false;
      submitBtn.textContent = 'إرسال';
    }
  });
})
</script>
